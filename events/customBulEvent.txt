#Fate of the Patriarchate
country_event = {
	id = bul_event.6
	title = bul_event.6.t
	desc = bul_event.6.d
	picture = 
	
	is_triggered_only = yes
	
	option = {
		name = bul_event.6.a
		highlight = yes
		trigger = {
			NOT = {
				patriarch_authority = 0.50
			}
		}
		add_country_modifier = {
			name = declining_patriarch
			duration = -1
		}
		if = {
			limit = {
				stability = 3
			}
			change_government_reform_progress = 100
		}
		else = {
			change_government_reform_progress = 50
		}
	}
	option = {
		name = bul_event.6.b
		highlight = yes
		trigger = {
			patriarch_authority = 0.50
			stability = 3
		}
		remove_country_modifier = freedom_people
		change_government_reform_progress = -50
	}
}
