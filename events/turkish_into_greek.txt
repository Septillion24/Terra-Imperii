namespace = turkish_into_greek



province_event = {
	id = turkish_into_greek.1
	title = "turkish_into_greek.2a"
	desc = "turkish_into_greek.2b"
	picture = POPE_PREACHING_eventPicture
	trigger = {
		owner = {
			OR = {
				culture_group = byzantine
			}
			has_country_flag = turkish_into_greek_flag
		}
		any_neighbor_province = {
			AND = {
				culture_group = byzantine
				NOT = { culture = turkish_greek }
			}
		}
		culture = turkish
	}
	mean_time_to_happen = {
		months = 60
	}	
	hidden = yes
	option = {
		name = "1"
		change_culture = turkish_greek
	}
}
